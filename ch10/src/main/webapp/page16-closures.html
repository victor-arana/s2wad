Closures!

        function makeAdder(x) {
            return function(n) {
                return n + x;
            }
        }

        var add10 = makeAdder(10);

        document.writeln(add10(1));  //  1 + 10 = 11
        document.writeln(add10(32)); // 32 + 10 = 42

        var add30 = makeAdder(30);

        document.writeln(add10(32)); // 32 + 10 = 42
        document.writeln(add30(12)); // 12 + 30 = 42
